def parse_receipt(text: str) -> list[dict]:
    lines = text.strip().split('\n')
    results = []
    
    for line in lines:
        if line.strip():
            line = line.replace(' x ', ' × ')
            parts = line.split(' × ')
            if len(parts) == 2:
                name = parts[0].strip()
                qty_price = parts[1].split(' @ ¥')
                if len(qty_price) == 2:
                    qty = int(qty_price[0].strip())
                    unit_price = float(qty_price[1].strip())
                    subtotal = qty * unit_price
                    results.append({"name": name, "qty": qty, "unit_price": unit_price, "subtotal": subtotal})
    
    return results
